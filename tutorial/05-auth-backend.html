Auth 백엔드 — 매직링크

physical-spark의 passwordless 매직링크 인증을 TypeScript로 이식합니다. SQLite 스키마 + /api/auth/start(링크 발송) + /api/auth/verify(세션 발급). dev에선 링크를 콘솔에 출력합니다.

import Database from "better-sqlite3";
import path from "node:path";
import fs from "node:fs";

const DB_PATH = process.env.PASTE_JD_DB || path.join(process.cwd(), "data", "paste-jd.db");
fs.mkdirSync(path.dirname(DB_PATH), { recursive: true });

// Next dev는 핫리로드마다 모듈을 재실행한다.
// globalThis에 캐싱해 연결이 한 번만 열리도록 한다(= "database is locked" 방지).
const g = globalThis as unknown as { _pasteDb?: Database.Database };
export const db = g._pasteDb ?? new Database(DB_PATH);
g._pasteDb = db;

db.pragma("journal_mode = WAL");

// physical-spark와 동일한 스키마. 타임스탬프는 정수 Unix초.
db.exec(`
  CREATE TABLE IF NOT EXISTS users(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    email TEXT UNIQUE NOT NULL,
    nickname TEXT UNIQUE NOT NULL,
    created_at INTEGER NOT NULL
  );
  CREATE TABLE IF NOT EXISTS login_tokens(
    token_hash TEXT PRIMARY KEY,
    email TEXT NOT NULL,
    expires_at INTEGER NOT NULL,
    used INTEGER NOT NULL DEFAULT 0,
    created_at INTEGER NOT NULL
  );
  CREATE TABLE IF NOT EXISTS sessions(
    id TEXT PRIMARY KEY,
    user_id INTEGER NOT NULL,
    expires_at INTEGER NOT NULL
  );
`);

2. 토큰 / 세션 / 닉네임 헬퍼

web/src/lib/auth.tstypescript
import crypto from "node:crypto";

export const TOKEN_TTL = 15 * 60;          // 매직링크 15분
export const SESSION_TTL = 30 * 24 * 3600; // 세션 30일
export const RESEND_COOLDOWN = 30;         // 이메일당 30초
export const COOKIE_NAME = "pr_session";

export const now = () => Math.floor(Date.now() / 1000);

// 256-bit CSPRNG. 토큰과 세션 id 모두 이걸로 생성.
export const newToken = () => crypto.randomBytes(32).toString("base64url");

// 저장은 항상 해시만. 원문 토큰은 이메일 링크에만 존재.
export const sha256 = (s: string) =>
  crypto.createHash("sha256").update(s).digest("hex");
web/src/lib/nicknames.tstypescript
import crypto from "node:crypto";

const ADJECTIVES = ["brave","calm","clever","bold","bright","swift","keen","lucky",
  "mellow","nimble","quiet","sunny","witty","zesty","cosmic","royal","amber","coral",
  "jade","noble","rapid","spry","vivid","wild","gentle","merry","plucky","snappy",
  "cheery","dandy"];
const NOUNS = ["otter","falcon","maple","comet","pixel","robot","gecko","koala","lynx",
  "puffin","quokka","tapir","walrus","yak","zebra","badger","cactus","dynamo","ember",
  "ferry","gizmo","harbor","island","jetpack","kettle","lantern","meteor","nugget",
  "orbit","piston"];

const pick = <T,>(a: T[]) => a[crypto.randomInt(a.length)];

// adjective-noun-NN (30 × 30 × 100 = 90,000). 중복이면 다시 뽑는다(rejection sampling).
export function generateNickname(taken: (n: string) => boolean): string {
  for (;;) {
    const n = `${pick(ADJECTIVES)}-${pick(NOUNS)}-${String(crypto.randomInt(100)).padStart(2, "0")}`;
    if (!taken(n)) return n;
  }
}

3. Route Handlers

Why /api/auth/* 같은 origin

auth API를 사이트와 같은 도메인에 두면 쿠키가 first-party가 되어 CORS·SameSite=None이 필요 없습니다. physical-spark이 "가장 중요한 아키텍처 결정"이라 부른 부분 — Next.js에선 app/api/auth/*에 두기만 하면 자동으로 지켜집니다.

터미널 · web/
mkdir -p src/app/api/auth/start src/app/api/auth/verify
web/src/app/api/auth/start/route.tstypescript
import { NextResponse } from "next/server";
import { db } from "@/lib/db";
import { now, newToken, sha256, TOKEN_TTL, RESEND_COOLDOWN } from "@/lib/auth";
import { sendMagicLink } from "@/lib/email";

export async function POST(req: Request) {
  const body = await req.json().catch(() => ({}));
  const email = String(body.email || "").toLowerCase().trim();

  // 계정 존재 여부를 노출하지 않도록 응답은 항상 동일.
  const ok = NextResponse.json({ ok: true, message: "Check your email for a sign-in link." });
  if (!email || !email.includes("@")) return ok;

  // 이메일당 30초 쿨다운
  const recent = db
    .prepare("SELECT created_at FROM login_tokens WHERE email=? ORDER BY created_at DESC LIMIT 1")
    .get(email) as { created_at: number } | undefined;
  if (recent && now() - recent.created_at < RESEND_COOLDOWN) return ok;

  const token = newToken();
  db.prepare(
    "INSERT INTO login_tokens(token_hash, email, expires_at, used, created_at) VALUES(?,?,?,0,?)"
  ).run(sha256(token), email, now() + TOKEN_TTL, now());

  const link = `${new URL(req.url).origin}/api/auth/verify?token=${token}`;
  sendMagicLink(email, link);
  return ok;
}
web/src/app/api/auth/verify/route.tstypescript
import { NextResponse } from "next/server";
import { db } from "@/lib/db";
import { now, sha256, newToken, SESSION_TTL, COOKIE_NAME } from "@/lib/auth";
import { generateNickname } from "@/lib/nicknames";

export async function GET(req: Request) {
  const token = new URL(req.url).searchParams.get("token") || "";
  const hash = sha256(token);

  const row = db
    .prepare("SELECT email, expires_at, used FROM login_tokens WHERE token_hash=?")
    .get(hash) as { email: string; expires_at: number; used: number } | undefined;

  if (!row || row.used || row.expires_at < now()) {
    return NextResponse.json(
      { error: "This sign-in link is invalid or has expired." },
      { status: 400 }
    );
  }

  // 단회용: 즉시 소각
  db.prepare("UPDATE login_tokens SET used=1 WHERE token_hash=?").run(hash);

  // find-or-create (로그인=회원가입 한 경로)
  let user = db.prepare("SELECT id FROM users WHERE email=?").get(row.email) as
    | { id: number }
    | undefined;
  if (!user) {
    const nick = generateNickname((n) => !!db.prepare("SELECT 1 FROM users WHERE nickname=?").get(n));
    const info = db
      .prepare("INSERT INTO users(email, nickname, created_at) VALUES(?,?,?)")
      .run(row.email, nick, now());
    user = { id: Number(info.lastInsertRowid) };
  }

  // 서버측 세션 발급
  const sid = newToken();
  db.prepare("INSERT INTO sessions(id, user_id, expires_at) VALUES(?,?,?)").run(
    sid,
    user.id,
    now() + SESSION_TTL
  );

  const res = NextResponse.redirect(new URL("/?welcome=1", req.url), 302);
  res.cookies.set(COOKIE_NAME, sid, {
    httpOnly: true,
    secure: process.env.NODE_ENV === "production", // localhost(http)에선 false여야 저장됨
    sameSite: "lax",
    domain: process.env.PASTE_JD_COOKIE_DOMAIN || undefined, // 배포 시 .bit-habit.com
    path: "/",
    maxAge: SESSION_TTL,
  });
  return res;
}
dev vs 배포 쿠키 차이

로컬은 http라 secure:true면 브라우저가 쿠키를 버립니다 → NODE_ENV로 분기. Domain도 로컬에선 설정하지 않아 host-only(localhost)로 동작하고, 배포에선 PASTE_JD_COOKIE_DOMAIN=.bit-habit.com을 줍니다(#8/#9).

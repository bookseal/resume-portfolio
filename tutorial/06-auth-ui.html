1. /me 와 /logout 라우트

터미널 · web/
mkdir -p src/app/api/auth/me src/app/api/auth/logout
web/src/app/api/auth/me/route.tstypescript
import { NextResponse } from "next/server";
import { cookies } from "next/headers";
import { db } from "@/lib/db";
import { now, COOKIE_NAME } from "@/lib/auth";

export async function GET() {
  const jar = await cookies();               // Next 15: cookies()는 async
  const sid = jar.get(COOKIE_NAME)?.value;
  if (!sid) return NextResponse.json({ error: "not signed in" }, { status: 401 });

  const row = db.prepare(
    `SELECT u.email, u.nickname, s.expires_at
       FROM sessions s JOIN users u ON u.id = s.user_id
      WHERE s.id = ?`
  ).get(sid) as { email: string; nickname: string; expires_at: number } | undefined;

  if (!row || row.expires_at < now()) {
    return NextResponse.json({ error: "not signed in" }, { status: 401 });
  }
  return NextResponse.json({ email: row.email, nickname: row.nickname });
}
web/src/app/api/auth/logout/route.tstypescript
import { NextResponse } from "next/server";
import { cookies } from "next/headers";
import { db } from "@/lib/db";
import { COOKIE_NAME } from "@/lib/auth";

export async function POST() {
  const jar = await cookies();
  const sid = jar.get(COOKIE_NAME)?.value;
  if (sid) db.prepare("DELETE FROM sessions WHERE id = ?").run(sid); // 서버측 진짜 무효화

  const res = NextResponse.json({ ok: true });
  res.cookies.set(COOKIE_NAME, "", {
    path: "/",
    domain: process.env.PASTE_JD_COOKIE_DOMAIN || undefined,
    maxAge: 0, // 즉시 삭제
  });
  return res;
}
Why 서버 세션 삭제

쿠키만 지우면 토큰이 유효한 채 남습니다. DELETE FROM sessions로 서버 기록을 없애야 “진짜 로그아웃”입니다. JWT였다면 만료까지 무효화가 안 됩니다 — 이게 서버 세션을 택한 이유.

2. 클라이언트 위젯

web/src/components/AuthWidget.tsxtsx
"use client"; // 상태/클릭이 브라우저에서 동작해야 하므로 클라이언트 컴포넌트

import { useEffect, useState } from "react";

type Me = { email: string; nickname: string };

export default function AuthWidget() {
  const [me, setMe] = useState<Me | null>(null);
  const [open, setOpen] = useState(false);
  const [email, setEmail] = useState("");
  const [msg, setMsg] = useState("");

  async function refresh() {
    const r = await fetch("/api/auth/me", { credentials: "include" });
    setMe(r.ok ? await r.json() : null);
  }
  useEffect(() => { refresh(); }, []); // 처음 붙을 때 한 번 로그인 상태 확인

  async function send() {
    if (!email.includes("@")) { setMsg("이메일을 확인하세요."); return; }
    const r = await fetch("/api/auth/start", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      credentials: "include",
      body: JSON.stringify({ email }),
    });
    setMsg(r.ok ? "✓ 메일함(개발 중엔 서버 콘솔)을 확인하세요." : "오류가 발생했습니다.");
  }

  async function logout() {
    await fetch("/api/auth/logout", { method: "POST", credentials: "include" });
    refresh();
  }

  if (me) {
    return (
      <div className="flex items-center gap-3 text-sm">
        <span>🎮 {me.nickname}</span>
        <button onClick={logout} className="text-neutral-500 hover:underline">logout</button>
      </div>
    );
  }

  return (
    <div className="text-sm">
      <button onClick={() => setOpen(true)}
        className="rounded-md bg-teal-600 px-3 py-1.5 font-medium text-white hover:bg-teal-700">
        Sign in
      </button>

      {open && (
        <div className="fixed inset-0 z-50 grid place-items-center bg-black/40"
             onClick={() => setOpen(false)}>
          <div className="w-80 rounded-xl bg-white p-5 dark:bg-neutral-900"
               onClick={(e) => e.stopPropagation()}>
            <h3 className="font-semibold">이메일로 로그인</h3>
            <p className="mt-1 text-xs text-neutral-500">비밀번호 없이 링크를 보냅니다.</p>
            <input
              type="email" value={email} placeholder="you@example.com"
              onChange={(e) => setEmail(e.target.value)}
              onKeyDown={(e) => e.key === "Enter" && send()}
              className="mt-3 w-full rounded-md border border-black/15 px-3 py-2
                         dark:border-white/20 dark:bg-neutral-800"
            />
            <button onClick={send}
              className="mt-3 w-full rounded-md bg-teal-600 py-2 font-medium text-white
                         hover:bg-teal-700">
              로그인 링크 보내기
            </button>
            {msg && <p className="mt-2 text-xs text-neutral-600 dark:text-neutral-300">{msg}</p>}
          </div>
        </div>
      )}
    </div>
  );
}
state 3개가 하는 일

me 로그인 상태 · open 모달 열림 여부 · email 입력값 · msg 안내문. setX(...)를 호출하면 React가 그 부분만 다시 그립니다. DOM을 직접 건드리지 않는 게 React 방식입니다.

3. 헤더에 위젯 넣기

모든 페이지 상단에 보이도록 layout.tsx에 헤더를 추가합니다. 기존 {children}은 그대로 두고 그 위에 헤더만 얹습니다.

web/src/app/layout.tsxtsx (body 안 구조)
import Link from "next/link";
import AuthWidget from "@/components/AuthWidget";
// ...기존 import(폰트/metadata)는 유지...

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="ko">
      <body>
        <header className="flex items-center gap-4 border-b border-black/10 px-6 py-3
                          dark:border-white/10">
          <Link href="/" className="font-bold">paste-jd</Link>
          <Link href="/cookbook" className="text-sm text-neutral-500">Cookbook</Link>
          <Link href="/dashboard" className="text-sm text-neutral-500">Dashboard</Link>
          <div className="ml-auto"><AuthWidget /></div>
        </header>
        {children}
      </body>
    </html>
  );
}
주의

create-next-app이 만든 원래 layout.tsx에는 폰트 변수와 metadata가 있습니다. 그 부분은 지우지 말고, <body> 내부 구조만 위처럼 바꾸세요. <body className={...}>의 기존 클래스도 유지하면 됩니다.

1. 재사용 세션 헬퍼

#6의 /me 로직을 서버 컴포넌트 어디서나 쓸 수 있게 함수로 뺍니다.

web/src/lib/session.tstypescript
import { cookies } from "next/headers";
import { db } from "@/lib/db";
import { now, COOKIE_NAME } from "@/lib/auth";

export type CurrentUser = { id: number; email: string; nickname: string };

// 로그인 사용자 or null. Server Component / Route Handler 어디서나 호출.
export async function getCurrentUser(): Promise<CurrentUser | null> {
  const sid = (await cookies()).get(COOKIE_NAME)?.value;
  if (!sid) return null;
  const row = db.prepare(
    `SELECT u.id, u.email, u.nickname, s.expires_at
       FROM sessions s JOIN users u ON u.id = s.user_id
      WHERE s.id = ?`
  ).get(sid) as (CurrentUser & { expires_at: number }) | undefined;
  if (!row || row.expires_at < now()) return null;
  return { id: row.id, email: row.email, nickname: row.nickname };
}
리팩터링 팁

#6의 /api/auth/me/route.ts도 이 헬퍼를 쓰도록 고치면 중복이 사라집니다. const u = await getCurrentUser(); return u ? NextResponse.json(u) : NextResponse.json({error:"not signed in"},{status:401});

4. 대시보드에 게이팅 적용

dashboard/page.tsx를 서버에서 사용자 확인 후 분기하도록 고칩니다.

web/src/app/dashboard/page.tsxtsx
import Link from "next/link";
import { DOCS } from "@/lib/documents";
import { getCurrentUser } from "@/lib/session";
import { isOwner } from "@/lib/owner";

export default async function Dashboard() {
  const user = await getCurrentUser();
  const owner = isOwner(user?.email);

  return (
    <main className="mx-auto max-w-3xl px-6 py-16">
      <h1 className="text-3xl font-bold">Dashboard</h1>

      {/* 누구의 데이터를 보는지 명확히 */}
      <div className="mt-3 inline-block rounded-full bg-neutral-100 px-3 py-1 text-sm
                      dark:bg-neutral-800">
        {owner ? `👤 ${user!.nickname} — 본인 데이터` : "🧪 John Doe — 공개 데모"}
      </div>

      {/* 소유자 전용 섹션 */}
      {owner ? (
        <section className="mt-8 rounded-xl border border-teal-500/40 bg-teal-500/10 p-5">
          <h2 className="font-semibold">본인 산출물</h2>
          <p className="mt-1 text-sm text-neutral-600 dark:text-neutral-300">
            실데이터는 서버의 개인 폴더에서 읽습니다(배포 후 scp로 이관 — #10).
            지금은 자리만 마련돼 있습니다.
          </p>
        </section>
      ) : (
        <p className="mt-6 text-sm text-neutral-500">
          🔒 소유자 계정으로 로그인하면 본인 데이터가 보입니다. 아래는 공개 데모입니다.
        </p>
      )}

      {/* 데모 문서는 항상 표시 */}
      <div className="mt-8 grid gap-4 sm:grid-cols-2">
        {DOCS.map((doc) => (
          <Link key={doc.slug} href={`/dashboard/${doc.slug}`}
            className="rounded-xl border border-black/10 bg-white p-5 hover:border-teal-500
                       dark:border-white/10 dark:bg-neutral-900">
            <h2 className="font-semibold">{doc.title}</h2>
            <p className="mt-1 text-sm text-neutral-500">
              {doc.fit.ok ? "✓ 한 페이지" : "⚠ 초과"} · {doc.fit.lines}/{doc.fit.budget}줄
            </p>
          </Link>
        ))}
      </div>
    </main>
  );
}
Why 서버에서 분기

이 페이지는 async Server Component라 await getCurrentUser()를 그대로 씁니다. 소유자가 아니면 소유자 섹션의 JSX 자체가 생성되지 않아 브라우저로 내려가지 않습니다. “클라이언트에서 display:none으로 숨기기”와는 보안 수준이 다릅니다.

완전 보호 라우트가 필요하면

어떤 페이지를 비로그인 시 아예 차단하려면 서버 컴포넌트 맨 위에서 if (!user) redirect("/")(next/navigation의 redirect). 미들웨어(middleware.ts)로 경로 단위 보호도 가능하지만, 지금 규모엔 컴포넌트 내 체크로 충분합니다.
